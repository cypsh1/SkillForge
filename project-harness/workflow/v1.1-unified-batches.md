**重写 `extractFrontmatter`** — 返回值增加 `status` 和 `rawFrontmatter`：

```typescript
export function extractFrontmatter(content: string): {
  frontmatter: SkillFrontmatter
  body: string
  status: FrontmatterStatus
  rawFrontmatter: string | null
} {
  const match = content.match(FRONTMATTER_REGEX)
  if (!match) {
    return {
      frontmatter: { name: "unknown" },
      body: content,
      status: "missing",
      rawFrontmatter: null,
    }
  }

  let frontmatter: SkillFrontmatter
  try {
    const parsed = parseYaml(match[1])
    if (!parsed || typeof parsed !== "object") throw new Error("not an object")
    frontmatter = parsed as SkillFrontmatter
  } catch {
    return {
      frontmatter: { name: "unknown" },
      body: match[2].trim(),
      status: "invalid",
      rawFrontmatter: match[1],
    }
  }

  if (!frontmatter.name) {
    frontmatter.name = "unknown"
  }

  return {
    frontmatter,
    body: match[2].trim(),
    status: "valid",
    rawFrontmatter: match[1],
  }
}
```

```typescript
export function parseSkillMd(
  content: string,
  skillId: string,
  skillPath: string = "",
): ParsedSkill {
  const { frontmatter, body, status, rawFrontmatter } = extractFrontmatter(content)
  const sections = extractSections(body)
  const tools = extractTools(frontmatter, sections)

    id: skillId,
    path: skillPath,

    rawFrontmatter,
    bodyDocument,
    description: frontmatter.description ?? extractLeadParagraph(body),
    tools,
    envVars,
    sections,
    rawContent: content,
    hasConfig: false,
    configFiles: {},
    extraFiles: {},
  }
}
```

```typescript
import type { FrontmatterStatus } from "@/types/skill"

export function serializeSkillMd(
  frontmatter: SkillFrontmatter,
  markdownBody: string,
  status?: FrontmatterStatus,
): string {
  if (status === "missing") {
    return markdownBody
  }
  const payload = buildFrontmatterPayload(frontmatter)
  const yamlText = stringify(payload, {
    lineWidth: 0,
    defaultStringType: "QUOTE_DOUBLE",
    defaultKeyType: "PLAIN",
  }).replace(/\n$/, "")

  return `---\n${yamlText}\n---\n${markdownBody}`
}
```

#### 5. 预览面板适配

`InspectorPanel` 中也需要根据 `frontmatterStatus` 决定预览策略：

- `missing`：只渲染 bodyDoc sections
- `invalid`：显示原始 frontmatter + bodyDoc sections
